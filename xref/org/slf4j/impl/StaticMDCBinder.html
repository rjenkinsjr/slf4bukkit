28  package org.slf4j.impl;
29  
30  import org.slf4j.helpers.BasicMDCAdapter;
31  import org.slf4j.spi.MDCAdapter;
32  
33  /**
34   * This implementation is bound to {@link BasicMDCAdapter}.
35   *
36   * @author Ceki G&uuml;lc&uuml;
37   */
38  public class StaticMDCBinder {
39  
40    /**
41     * The unique instance of this class.
42     */
43    public static final StaticMDCBinder SINGLETON = new StaticMDCBinder();
44  
45    private StaticMDCBinder() {
46    }
47  
48    /**
49     * Return the singleton of this class.
50     *
51     * @return the StaticMDCBinder singleton
52     * @since 1.7.14
53     */
54    public static final StaticMDCBinder getSingleton() {
55      return StaticMDCBinder.SINGLETON;
56    }
57  
58    /**
59     * Currently this method always returns an instance of {@link BasicMDCAdapter}
60     * .
61     *
62     * @return never null.
63     */
64    public MDCAdapter getMDCA() {
65      // note that this method is invoked only from within the static initializer
66      // of the org.slf4j.MDC class.
67      return new BasicMDCAdapter();
68    }
69  
70    public String getMDCAdapterClassStr() {
71      return BasicMDCAdapter.class.getName();
72    }
73  
74  }
